部署Mysql MHA集群模式

    1. 所有节点配置免密
$ ssh-keygen  
$ ssh-copy-id root@IP地址
    1. 所有节点安装所需服务
$ yum -y install mysql-community-*
$ yum -y install perl-DBD-mysql
  • 2.1 在所有服务器上安装MHA_node节点
$ rpm -ivh mha4mysql-node-0.56-0.el6.noarch.rpm
    1. 主库、备用主库开启半同步复制
  • 3.1 查看当前数据库是否支持动态加载模块
mysql> show variables like 'have_dynamic_loading';
+----------------------+-------+
| Variable_name        | Value |
+----------------------+-------+
| have_dynamic_loading | YES   |
+----------------------+-------+
1 row in set (0.02 sec)
  • 3.2 命令安装模块
mysql> install plugin rpl_semi_sync_master soname 'semisync_master.so';  	# 主模块
mysql> install plugin rpl_semi_sync_slave soname 'semisync_slave.so';  	# 从模块
  • 3.3 查看模块状态
mysql> select plugin_name,plugin_status from information_schema.plugins where plugin_name like '%semi%';
+----------------------+---------------+
| plugin_name          | plugin_status |
+----------------------+---------------+
| rpl_semi_sync_master | ACTIVE        |
| rpl_semi_sync_slave  | ACTIVE        |
+----------------------+---------------+
2 rows in set (0.00 sec)
  • 3.4 启动半同步复制
mysql> set global rpl_semi_sync_master_enabled=1;  	# 主模块
Query OK, 0 rows affected (0.00 sec)

mysql> set global rpl_semi_sync_slave_enabled=1;  	# 从模块
Query OK, 0 rows affected (0.00 sec)

# 查看是否成功
mysql> show variables like 'rpl_semi_sync_%_enabled';
+------------------------------+-------+
| Variable_name                | Value |
+------------------------------+-------+
| rpl_semi_sync_master_enabled | ON    |
| rpl_semi_sync_slave_enabled  | ON    |
+------------------------------+-------+
2 rows in set (0.00 sec)
  • 3.5 在my.cnf文件添加以下内容后保存退出重启数据库
$ cat >/etc/my.cnf<<-EOF
# 开启插件加载半同步复制
plugin-load="rpl_semi_sync_master=semi_sync_master.so;rpl_semi_sync_slave=semisync_slave.so"
rpl-semi-sync-master-enabled=1
rpl-semi-sync-slave-enabled=1

# 不允许自动删除中继日志文件,MySQL>set global relay_log_purge=off;
relay_log_purge=off; 

# 主服务器id、开启bin-log
server_id=1
log-bin=master_1
binlog-format="mixed"
EOF
    1. 登录主、备主数据库授权
mysql> grant replication slave on *.* to slave@'10.14.23.%' identified by '123456';
Query OK, 0 rows affected, 1 warning (10.00 sec)

mysql> show master status;
+-----------------+----------+--------------+------------------+-------------------+
| File            | Position | Binlog_Do_DB | Binlog_Ignore_DB | Executed_Gtid_Set |
+-----------------+----------+--------------+------------------+-------------------+
| master_1.000001 |      447 |              |                  |                   |
+-----------------+----------+--------------+------------------+-------------------+
1 row in set (0.00 sec)
  • 4.1 在备主数据库做以下同步操作
mysql> change master to 
    -> master_host='主库IP',
    -> master_user='slave',
    -> master_password='123456',
    -> master_log_file="master_1.000001",
    -> master_log_pos=447;
Query OK, 0 rows affected, 2 warnings (0.01 sec)

mysql> start slave;
Query OK, 0 rows affected (0.01 sec)

mysql> show slave status\G;
             Slave_IO_Running: Yes
            Slave_SQL_Running: Yes
  • 4.2 从服务器my.cnf配置
$ cat >/etc/my.cnf<<-EOF
server_id=3
EOF
  • 4.3 从MySQL配置
mysql> change master to 
    -> master_host='10.14.23.32',
    -> master_user='slave',
    -> master_password="123456",
    -> master_log_file="master_1.000001",
    -> master_log_pos=447;
Query OK, 0 rows affected, 2 warnings (0.01 sec)

mysql> start slave;

             Slave_IO_Running: Yes
            Slave_SQL_Running: Yes
  • 4.4 都配置完成后在主数据库上做授权
mysql> grant all on *.* to root@'10.14.23.%' identified by '123456';
Query OK, 0 rows affected, 1 warning (0.00 sec)
    1. 都配置完成在所有服务器运行以下命令
$ cd mha-soft-student/
$ yum -y install perl-*.rpm
$ yum -y install perl-DBD-mysql
  • 5.1 在管理主机运行
$ yum -y install perl-ExtUtils- perl-CPAN-*
[root@localhost mha-soft-student]# tar -zxvf mha4mysql-manager-0.56.tar.gz
[root@localhost mha-soft-student]# cd mha4mysql-manager-0.56/
[root@localhost mha-soft-student]# perl  Makefile.pl
[root@localhost mha-soft-student]# make
[root@localhost mha-soft-student]# make  install
  • 5.2 在管理主机创建命令
[root@localhost mha4mysql-manager-0.56]# mkdir /root//bin
[root@server mha4mysql-manager-0.56]# cp bin/* /root/bin/
  • 5.3 创建主配置 并编辑
[root@localhost bin]# mkdir /etc/mha
[root@server mha4mysql-manager-0.56]# cp samples/conf/app1.cnf /etc/mha/
  • 5.4 修改.cnf配置文件
[root@localhost mha-soft-student]# cat >/etc/mha/app1.cnf<<-EOF
[server default]  #服务默认配置
manager_workdir=/etc/mha   #工作目录
manager_log=/etc/mha/manager.log #日志文件
#故障切换脚本
master_ip_failover_script=/etc/mha/master_ip_failover

##主数据库bin-log所在目录
master_binlog_dir=/usr/local/mysql/data/

#ssh服务用户名及端口
ssh_user=root
ssh_port=22

#主从复制时，从库连接主库的用户名及密码
repl_user=slave
repl_password=123456

#监控数据库时，连接服务器的用户及密码
user=root
password=123456

##指定数据库服务器ip及角色
[server1]
hostname=192.168.4.61
candidate_master=1
port=3306

[server2]
hostname=192.168.4.62
candidate_master=1
port=3306

[server3]
hostname=192.168.4.65
no_master=1
port=3306
EOF
    1. 创建故障切换脚本并编辑
[root@server ~]# cd mha-soft-student
[root@server mha-soft-student]# cp master_ip_failover /etc/mha/
[root@server ~]# chmod +x /etc/mha/master_ip_failover
  • 6.1 修改脚本
$ cat >/etc/mha/master_ip_failover<<-EOF
 35 my $vip = '192.168.4.100/24';  # Virtual IP 
 36 my $key = "1";
 37 my $ssh_start_vip = "/sbin/ifconfig eth0:$key $vip";
 38 my $ssh_stop_vip = "/sbin/ifconfig eth0:$key down";
 EOF
  • 6.2 在主库上配置脚本里定好的IP
[root@localhost ~]# ifconfig ens32:1 10.14.23.253/24
[root@localhost ~]# ifconfig ens32:1
ens32:1: flags=4163<UP,BROADCAST,RUNNING,MULTICAST>  mtu 1500
        inet 10.14.23.253  netmask 255.255.255.0  broadcast 10.14.23.255
        ether 00:0c:29:0b:a3:12  txqueuelen 1000  (Ethernet)
    1. 启动服务、检查ssh连接
[root@server mha-soft-student]# masterha_check_ssh --conf=/etc/mha/app1.cnf
Wed Aug 28 11:38:59 2019 - [info] All SSH connection tests passed successfully.  	# 说明正确
  • 7.1 检查主从同步配置
[root@server mha-soft-student]# masterha_check_repl --conf=/etc/mha/app1.cnf
MySQL Replication Health is OK.  	# 说明成功

ln -s /usr/local/mysql/bin/mysqlbinlog /usr/bin/mysqlbinlog  	# 报错需在所有服务器上做链接
  • 7.2 启动管理服务
[root@server bin]# masterha_manager --conf=/etc/mha/app1.cnf  --remove_dead_master_conf --ignore_last_failover
  • 7.3 查看状态
[root@server ~]# masterha_check_status --conf=/etc/mha/app1.cnf
app1 (pid:60184) is running(0:PING_OK), master:10.14.23.32
    1. 测试配置_户端上连接vip 192.168.4.100访问数据库服务

  • 8.2 客户端使用授权用户连接
[root@server ~]# mysql -h10.14.23.253 -uroot -p
mysql> insert into db1.a (id, name) values(101,"wang");

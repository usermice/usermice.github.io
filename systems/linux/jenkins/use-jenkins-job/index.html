##!/bin/bash
path=$JENKINS_HOME/workspace/$JOB_NAME
echo $BUILD_ID
echo $path
git reset --hard $git
git reset --hard  $endid
/opt/script/online.sh  $BUILD_ID  $path

rm -rf diff_files.tx*
tar tf diff_online.tar.gz > diff_files.txt

declare -A dict
dict=(
 [FILES]="$(tar tf diff_online.tar.gz |xargs)"
 [P_NAME]="ansible-hosts"
 [I_NAME]="project"
 [I_PATH]="/opt/data/project"
 [B_PATH]="/opt/backup/"
)

ansible ${dict['P_NAME']} -m shell \
-a "cd ${dict['I_PATH']} && tar czf ${dict['B_PATH']}/${dict['I_NAME']}-$BUILD_NUMBER.tar.gz --files-from ${dict['FILES']} 2> /dev/null; echo backup codes"

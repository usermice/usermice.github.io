  • 1.3 配置sonar-scanner的sonar-project.properties文件

$ cat >/opt/php/project/sonar-project.properties<<-EOF
# 必须指定projectKey
sonar.projectKey=php:admin
# 定义检测后展示名称
sonar.projectName=Project NAME
# 指定版本
sonar.projectVersion=4.7 
# 项目代码存放目录 
sonar.sources=/opt/admin
# 项目代码使用的开发语言
sonar.language=php 
sonar.dynamicAnalysis=false 
# Encoding of the source files 
sonar.sourceEncoding=UTF-8
EOF

# 切换项目目录下执行
$ /opt/sonar-scanner/bin/sonar-scanner -Dsonar.login=admin -Dsonar.password=password  -Dsonar.host.url=http://sonarqube.erge.com

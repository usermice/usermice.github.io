expect使用

  • **Expect**是一个自动化工具，用于与交互式程序进行交互。它允许你编写脚本来模拟人类用户与程序的交互过程，从而自动化执行那些通常需要人工干预的任务
  • 主要特点和用途：

  • 自动化交互：

    Expect脚本可以编写为模拟用户输入和处理程序输出的行为。它通过捕获和分析程序的输出，并根据预定义的规则自动响应，来执行所需的操作。

  • 与交互式程序集成：

    适用于需要用户输入或期望特定响应的程序，例如配置管理工具、网络设备配置、数据库管理等。

  • 简化复杂任务：

    可以通过Expect自动化执行复杂的系统管理任务，如安装和配置软件、执行备份和恢复操作、模拟用户会话等。

  • 基于TCL语言：

    Expect的脚本语言基于TCL（Tool Command Language），这使得它结合了TCL的强大编程能力和Expect提供的交互式自动化功能。

  • 使用场景：

  • 系统管理和自动化运维：

    批量配置和管理服务器和网络设备。

  • 测试和仿真：

    自动化测试脚本，模拟用户行为以测试程序的交互性。

  • 应用程序部署：

    在多个环境中自动化部署和配置应用程序。

  • 远程管理：

    通过SSH或Telnet等协议进行远程管理和操作。

  • 示例：
#!/usr/bin/expect

set username "your_username"
set password "your_password"
set hostname "remote_host"

spawn ssh $username@$hostname
expect "password:"
send "$password\r"
expect "$ "
send "ls -l\r"
expect "$ "
send "exit\r"
expect eof
$ cat >expect.sh<<-EOF
#!/use/bin/expect
spawn ssh  root@192.168.1.1  	# spawn:开启一个会话

expect {
"yes/no" { send "yes\r"; exp_continue }  	# send:传送参数，\r:回车键, exp_continue:没出现"yes/no"也提交 第二次连接不会出现，
"password:" { send "密码\r" };
}
interact  	# 停留在被连接系统中
EOF

-ne、!= 不等于： 检测两个数是否不相等，不相等返回true [ $a -ne $b ] 返回：true
-gt、> 大于：检测左边的数是否大于右边的数，如果是返回：true [ $a -gt $b ] 返回：false
-lt、< 小于：检测左边的数是否小于右边的，如果是返回：true [ $a -lt $b ] 返回：true

$ echo '#!/usr/bin/bash
# 提示用户没有加参数
if [ $# -eq 0 ]; then
 echo "执行方式: `basename $0` 文件名(file)"
 exit  # 退出
fi

# 提示输入的不是文件: ! 取反:不是文件
if [ ! -f $1 ];then
 echo "错误的文件名"
 exit
fi

# 正确执行
for IP in `cat $1`
do 
 ping -c1 $IP &>/dev/null
 if [ $? -eq 0 ];then
     echo "${IP} is up"
 else
     echo "${IP} is down"
 fi
done
' >ping.sh

$ ./ping.sh ip.txt

$ echo '#!/usr/bin/bash
read -p "请输入数字: " num
while true
do
    if [[ "$num" =~ ^[0-9]+$ ]];then
        break   # 跳出循环
    else
        echo "你输入的不是数字！！！"
        read -p "输入错误，请重新输入: " num
    fi
done

read -p "请输入创建的用户首字母: " prefix

if [ -z "$prefix" ];then
    echo "错误的首字母(prefix)"
    exit
fi

for i in `seq $num`
do
    user=$prefix$i
    useradd $user
    echo "123" |passwd --stdin $user &>/dev/null
    if [ $? -eq 0 ];then
        echo "$user 创建成功"
    fi
done  ' >useradd.sh

	case $ID in
		0)

		 read  cache
		 if [ -d ${directory}/${Name}/${cache}/ ];then
			/usr/bin/ansible web -i /etc/ansible/hosts -m shell  -a "rm -rf ${directory}/${Name}/${cache}" -f 5 &>/dev/null
			echo "已清理租户${Name}目录下${cache}目录中的缓存"
			exit
		 else
			echo "目录不存在"
			exit
		 fi;;

		2)
		 echo -en "\e[1;36m请输入要清理缓存的租户目录: \e[0m"

	 	 if [ -d ${directory}/${Name} ];then
			/usr/bin/ansible web -i /etc/ansible/hosts -m shell  -a "rm -rf ${directory}/${Name}/*" -f 5 &>/dev/null
			echo "已清理租户${Name}目录下所有缓存"
			exit
		 fi;;

		3)
		 echo -en "\e[1;31m确认要清理所有租户缓存吗[y/n]: \e[0m"
		 read y
		 if [ "$y" = "y" ];then
			/usr/bin/ansible web -i /etc/ansible/hosts -m shell  -a "rm -rf ${directory}/*" -f 5 &>/dev/null
			echo "已清理所有租户缓存"
			for mk in `cat /opt/data/Tenant.txt`
			do
				/usr/bin/ansible web -i /etc/ansible/hosts -m shell  -a "/usr/bin/mkdir -pv ${directory}/${mk} >>${directory}/directory.log" -f 5 &>/dev/null
				if [ $? -eq 0 ];then
					echo "${mk} create ok"
				else
					echo "${mk} create error"
				fi
			done
			exit
		else
			echo "输入不是’y‘退出"
			exit
		fi;;
	esac
done

$ cat >/etc/ssh/sshrc<<-EOF
#!/bin/sh
TELEGRAM_TOKEN=""
TELEGRAM_GROUP_ID=""

# 获取用户、IP地址、时间和服务器信息
USER_NAME=$USER
HOST_NAME=$(hostname -s)
DATA_TIME=$(date +"%F %H:%M:%S")
LOCALHOST=$(hostname -I | awk '{print $1}')
LOGGIN_IP=$(echo $SSH_CLIENT | awk '{print $1}')

# 构造通知消息
msg="有白名单之外的 IP 登录服务器，请验证:
User: ${USER_NAME}
Host: ${LOCALHOST}
Login IP: ${LOGGIN_IP}
Login Time: ${DATA_TIME}
Host Name: ${HOST_NAME}"

# 定义信任的IP地址列表
IP_WhiteList='192.168.1.100'
result=$(echo ${IP_WhiteList} | grep "${LOGGIN_IP}")

if [ "${result}" != "" ]; then
        exit 0  # 如果在信任列表中，退出脚本
else
        curl -X POST "https://api.telegram.org/bot${TELEGRAM_TOKEN}/sendMessage" -d "chat_id=${TELEGRAM_GROUP_ID}&text=${msg}" > /dev/null
fi
EOF

使用Fikker部署CDN

    1. 使用脚本部署Fikker缓存服务
  • 防火墙开放：80、443、6780等端口，6780：端口为管理端口，默认密码：123456
$ cat >/opt/Install-Fikker-cache.sh<<-EOF
##!/bin/bash

DOWNLOAD=/usr/local/src/
FikkerInstallDir="/opt/" # default installation directory
FikkerNewVersion="fikkerd-3.8.3-linux-x86-64" # package name

mkdir -p ${FikkerInstallDir}

if [ -f "/usr/bin/apt-get" ];then 
	apt-get install -y wget tar; 
fi
if [ -f "/usr/bin/yum" ];then 
	yum install -y wget tar; 
fi

wget -c --no-check-certificate -P ${DOWNLOAD} https://www.fikker.com/dl/$FikkerNewVersion.tar.gz

tar zxf ${DOWNLOAD}${FikkerNewVersion}.tar.gz -C ${FikkerInstallDir}

cd ${FikkerInstallDir}${FikkerNewVersion}
./fikkerd.sh install 
./fikkerd.sh start 
cd ${FikkerInstallDir} 
sleep 5 
echo 'finished!'
EOF

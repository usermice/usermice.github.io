配置Ansible-Playbook

    1. Ansible的内置模块列表
    1. 关于become属性
    1. Ansible-Playbook目录定义
ansible-playbook/
└── roles  # 剧本
    ├── mysql
    │   ├── files
    │   ├── handlers
    │   ├── tasks
    │   ├── templates
    │   └── vars
    ├── nginx  	# Nginx的角色剧本
    │   ├── files  	# 要用到的拷贝文件目录
    │   │   └── index.html
    │   ├── handlers  	# 触发器目录
    │   │   └── main.yaml
    │   ├── tasks  	# 任务主要目录，用来安装程序服务
    │   │   └── main.yaml
    │   ├── templates  	# 模板目录
    │   │   └── nginx.conf.j2
    │   └── vars  	# 变量目录
    │       └── main.yaml
    └── site.yaml  	# 定义那个节点或者那组服务器执行，可以定义多个角色剧本
    1. Ansible-playbook字段定义
  • 2.1 name：每个play名字
  • 2.2 hosts：每个play涉及被管理的服务器，同ad-hoc中的资产选择器
  • 2.3 tasks：每个play中具体要完成的任务，以列表的形式表达
  • 2.4 become：如果需要提权，则加上become相关属性
  • 2.5 become_user：若要提权的话，提权到那个用户上
  • 2.6 remote_user：指定连接到远程节点的用户，就是在远程服务器上执行具体操作的用户吗，若不指定，则默认使用当前执行ansible-playbook的用户

Ansible-Playbook→Roles→Nginx

    1. Roles→Nginx目录下文件夹配置
  • 1.1 files要用到的文件目录
$ cat >/opt/ansible-playbook/roles/nginx/files/index.html<<-EOF
TEST
EOF
  • 1.2 handlers触发器目录，
$ cat >/opt/ansible-playbook/roles/nginx/handlers/main.yaml<<-EOF
---
- name: restart nginx
  service: name=nginx state=restarted
EOF
  • 1.3 tasks任务主要目录，用来安装程序服务
$ cat >/opt/ansible-playbook/roles/nginx/tasks/main.yaml<<-EOF
--- # 声明
# 定义执行项目名称，安装服务
- name: Install Nginx packge
  yum: name={{ item }} state=lates # item 为内置变量
  with_items:
  - epel-release
  - nginx

# 拷贝文件
- name: Copy index.html
  copy: src=index.html dest=/usr/local/nginx/html/index.html

# 拷贝文件
- name: Copy nginx.conf Teplate
  temlate: src=nginx.conf.j2 dest=/usr/local/nginx/conf/
  notify: restart nginx  # nginx.conf.j2 文件拷贝成功，通知handlers(触发器) 重启使之生效

# 启动服务
- name: make sure nginx service running
  service: name=nginx state=started enabled=yes
EOF
  • 1.4 templates模板目录
$ cat >/opt/ansible-playbook/roles/nginx/templates/nginx.conf.j2<<-EOF
user  nobody;
worker_processes {{ ansible_processor_cores }};  	# 服务内置变量

events {
    worker_connections {{ worker_connections }};  	# 自定义变量
}

http {
    include       mime.types;
    default_type  application/octet-stream;

    sendfile        on;
    keepalive_timeout  65;

    server {
        listen       80;
        server_name  localhost;

        location / {
            root   html;
            index  index.html index.htm;
        }

        error_page   500 502 503 504  /50x.html;
        location = /50x.html {
            root   html;
        }
    }
}
EOF
  • 1.5 vars自定义变量目录
$ cat >/opt/ansible-playbook/roles/nginx/vars/main.yaml<<-EOF
worker_connections: 10240;
EOF

Ansible-Playbook→Roles→Site.yaml

    1. 配置Ansible-Playbook下Roles目录下site.yaml文件
  • 1.1 site.yaml文件，定义那个节点或者那组服务器执行什么程序
$ cat >/opt/ansible-playbook/roles/site.yaml<<-EOF
- hosts: web
  roles:
  - nginx

- hosts: mysql
  roles:
  - mysql
EOF
    1. 检查Ansible-Playbook语法以及执行
$ ansible-playbook /opt/ansible-playbook/roles/site.yaml --syntax-chek  	# 检查脚本语法 --syntax-chek 参数
$ ansible-playbook /opt/ansible-playbook/roles/site.yaml -C  	# -C 模拟运行查看流程是否会出错

$ ansible-playbook /opt/ansible-playbook/roles/site.yaml

1.1.3 Handlers触发器

$ cat >/opt/ansible-playbook/nginx.yaml<<-EOF
- hosts: web_server
 vars:
    app_path: "{{ base_path }}/app"
    http_port: 80
    max_clients: 256
  become: true  	# 需要目标主机的用户具有sudo权限
  tasks:
  - name: ensure apache is at the latest version
    yum: name=httpd state=latest
  - name: ensure apahech is running
    service: name=httpd state=started
  - name:  	# 创建或修改文件&目录
    file:
      path: /tmp/demo
      state: directory
      mode: 0755
  - name:  	# 拷贝文件到目标主机
    copy:
      src: /etc/hostname
      dest: /tmp/demo/hostname
  - name: disable selinux
    command: /sbin/setenforce 0
  - name: run this command and ignore the result
    shell: /usr/bin/somecommand || /bin/true 
    ignore_errors: True  	# 忽略报错信息
  - name:  	# 用模板生成配置文件
    template:
      src: demo.j2
      dest: "{{ ansible_user_dir }}/demo.cfg"
  handlers
    - name: restart apache
      service: name=httpd state=restart
EOF

$ cat <<EOF >roles/demo-role/tasks/main.yml
- include: yum.yml
  when: (ansible_distribution|lower == 'redhat') or (ansible_distribution|lower == 'centos')
- include: apt.yml
  when: (ansible_distribution|lower == 'debian') or (ansible_distribution|lower == 'ubuntu')
EOF

$ cat <<EOF >roles/demo-role/tasks/yum.yml
- name:  	# 使用Yum安装Apache服务
  yum:
    name: "{{ redhat_app }}"
    state: "{{ app_state }}"
  notify:
  - done
EOF

$ cat <<EOF >roles/demo-role/tasks/apt.yml
- name:  	# 使用Apt安装Apache服务
  apt:
    name: "{{ debian_app }}"
    state: "{{ app_state }}"
  notify:
  - done
EOF

$ cat <<EOF >roles/demo-role/vars/main.yml
redhat_app: httpd
debian_app: apache2
EOF

$ cat <<EOF >roles/demo-role/handlers/main.yml
- name: done
  debug:
    msg: "Apache服务已部署"
EOF
